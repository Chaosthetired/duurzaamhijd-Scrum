CREATE TABLE public.company_table (
	company_id serial4 NOT NULL,
	company_name varchar NULL,
	company_type int4 NULL,
	company_emissions numeric NULL,
	logo_id int4 NULL,
	company_source varchar NULL,
	company_electricity_use numeric NULL,
	company_version int4 NULL,
	company_status varchar NULL,
	company_user_submit_id int4 NULL,
	company_admin_reviewed_by_id int4 NULL,
	company_reviewed_at timestamp NULL,
	company_company_submited_at timestamp NULL,
	CONSTRAINT company_table_pk PRIMARY KEY (company_id)
);
